Introduction¶

Sobre o dataset¶

Descrio das colunas

C/A,UNIT,SCP,STATION,LINENAME,DIVISION,DATE,TIME,DESC,ENTRIES,EXITS

C/A      = Agrupamento de catracas de que a catraca faz parte (_Control Area_)
UNIT     = Cabine de controle associada  estao onde a catraca se encontra (_Remote Unit for a station_)
SCP      = Endereo especfico da catraca (_Subunit Channel Position_)
STATION  = Nome da estao onde a catraca se encontra
LINENAME = Cdigo representando todas linhas que passam na estao*
DIVISION = Cdigo representando a concessionria original da linha, antes da prefeitura assumir a gesto   
DATE     = Representa a data (no formato MM-DD-YY) do evento de auditoria agendado
TIME     = Representa o horrio (hh:mm:ss) do evento de auditoria agendado
DESc     = Descreve o tipo de evento de auditoria registrado:
           1. "REGULAR" representando um evento de auditoria padro, em que a contagem  feita a cada 4 horas
           2. "RECOVR AUD" significa que o valor especfico estava perdido, mas foi recuperado posteriormente 
           3. Diversos cdigos sinalizam situaes em que auditorias so mais frequentes devido a atividades de
              planejamento ou soluo de problemas. 
ENTRIES  = A contagem cumulativa de entradas associadas  catraca desde o ltimo registro
EXITS    = A contagem cumulativa de sadas associadas  catraca desde o ltimo registro

*  Normalmente as linhas so representadas por um caractere. LINENAME 456NQR significa que os trens 4, 5, 6, N, Q e R passam pela estao.

def load_dataset(sufix_url, base_url = "http://web.mta.info/developers/"):
    """
    Funo utilizada para baixar os arquivos de dados do metr de New York.
      Recebe a url base e o final da url, pois assim temos uma forma mais genrica de trabalhar
      com os dados. Caso mude a url, basta passar via parmetro.
      ----------------------------------------------------------
      sufix_url: final da url, o link do documento.
      base_url: a url do site, no caso o default  http://web.mta.info/developers/
    """
    
    u = urllib.urlopen(base_url+sufix_url)
    html = u.read()
    soup = BeautifulSoup(html, "html.parser")
    links = soup.find_all('a')
    for link in links:
        try:
            if link and '1706' in link.get('href'):
                filename = link.get('href').split('/')
                filename = filename[len(filename) - 1]
                urllib.urlretrieve(base_url + link.get('href'), filename)
                print("Lendo arquivo de {}. \nSalvando arquivo como {}"
                      .format(base_url + link.get('href'), filename))
        except:
            #Caso ocorra uma excesso lendo o arquivo do link(ex: 404), ignora e continua para o prximo
            continue

def create_master_turnstile_file(filenames, output_file):
    """
        Usado para unificar mais de um arquivo em um nico, pula a primeira linha de cada arquivo.
        Recebe uma lista de nomes de arquivos e o nome do arquivo de sada. 
        -------------------------------------------------------------------
        filenames: array de nomes de arquivos.
        output_file: o nome do arquivo de sada, onde vo ser salvo os dados dos arquivos de entrada.
    """
    with open(output_file, 'w') as master_file:
        master_file.write('C/A,UNIT,SCP,DATEn,TIMEn,DESCn,ENTRIESn,EXITSn\n')
        for filename in filenames:
            f = open(filename,"r")
            if f.mode == 'r':
                data = f.readlines()[1:]
                for line in data:
                    master_file.write(line)
                f.close()

def get_rows_difference(df, column, new_column, fill = 0):
    """
        Funo utilizada para pegar a diferena entre a linha posterior e linha anterior
        de um dado Dataframe.
        -------------------------------------------------------------------
        df: o Dataframe com os dados a serem analisados.
        column: coluna que ser calculada.
        new_column: o nome da nova coluna, gerada a partir do clculo da coluna informada.
        fill: valor para substituir registros com NaN
    """
    aux = df.copy()
    aux[new_column] = aux[column].shift()
    aux[new_column] = aux[column] - aux[new_column]
    aux[new_column].fillna(fill, inplace=True)
    return aux

def time_to_hour(time, pattern = '\d{2}:\d{2}:\d{2}', extract = '\d{2}'):
    """
        Funo utilizada para pegar converter uma String em formato de tempo para hora.
        -------------------------------------------------------------------
        time: a string a ser decomposta.
        pattern: a regex da string informada. Default:\d{2}:\d{2}:\d{2}
        extract: a regex a ser extrada da string. Default: \d{2}
    """
    
    result = re.match(pattern, time)   
    if result and result.group(): 
        aux = re.search(extract, time)
        if aux.group():
            return(int(aux.group(0)))
    raise Exception('String {} don\'t have the same pattern as {} or {} couldn\'t be found.'
                    .format(time, pattern, extract))

import sys
import codecs
sys.stdout = codecs.getwriter('utf8')(sys.stdout)

def mapper():
    first_line = True
    for line in sys.stdin:  
        line = line.decode('iso8859-1')
        if first_line:
            first_line = False
            continue
        aux = line.split(",")
        if len(aux) > 6:
            print("{}\t{}".format(aux[1], aux[6]).encode('utf-8'))  
        

sys.stdout = open('mapper_result.txt', 'w')
